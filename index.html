/** * WorkPulse — Google Apps Script Backend v2 * ════════════════════════════════════════════ * Full bi-directional sync between WorkPulse HTML and Google Sheets. * Data is stored in TWO ways simultaneously: * * 1. RAW (Sheet: "WP_Raw") — Full JSON blob, used for load/save * 2. READABLE (separate sheets) — Human-readable rows for each collection: * • WP_Users — All employee records * • WP_Attendance — Daily punch in/out records * • WP_Leaves — Leave requests and decisions * • WP_Ideas — Ideas submitted by team * • WP_SalarySlips — Salary slip metadata * • WP_Config — System settings (holidays, occasions, shift timings) * * SETUP (5 minutes): * ────────────────── * 1. Open Google Sheets → create a new blank spreadsheet * 2. Extensions → Apps Script → delete default code → paste this file * 3. Save → Deploy → New Deployment * Type: Web App * Execute: Me * Access: Anyone * 4. Click Deploy → COPY the Web App URL * 5. In WorkPulse.html find: const GAS_URL = ''; * Paste your URL inside the quotes and save * 6. Upload to GitHub Pages — done! * * HOW SYNC WORKS: * ─────────────── * Save → POST {action:'save'} → writes WP_Raw blob + all readable sheets * Load → GET ?action=load → reads WP_Raw and returns JSON to browser * Reset → POST {action:'reset'}→ clears WP_Raw (readable sheets kept as archive) * Auto-pull every 30s (multi-user live sync — latest timestamp wins) * * © 2025 Sanyam Rastogi. All rights reserved. */ const RAW_SHEET = 'WP_Raw'; const SHEETS = { users: 'WP_Users', attendance: 'WP_Attendance', leaves: 'WP_Leaves', ideas: 'WP_Ideas', salarySlips: 'WP_SalarySlips', config: 'WP_Config', }; function _cors(out) { return out .setHeader('Access-Control-Allow-Origin', '*') .setHeader('Access-Control-Allow-Methods', 'GET, POST, OPTIONS') .setHeader('Access-Control-Allow-Headers', 'Content-Type'); } function _sheet(name, headers) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sh = ss.getSheetByName(name); if (!sh) { sh = ss.insertSheet(name); if (headers && headers.length) { sh.getRange(1,1,1,headers.length).setValues([headers]) .setBackground('#1a1f3a').setFontColor('#ffffff') .setFontWeight('bold').setFontSize(10); sh.setFrozenRows(1); headers.forEach((_,i) => sh.setColumnWidth(i+1, i===4?280:150)); } } return sh; } function _clearData(sh, cols) { if (sh.getLastRow() > 1) sh.getRange(2, 1, sh.getLastRow()-1, cols).clearContent().clearFormat(); } function _writeRaw(db) { const sh = _sheet(RAW_SHEET, ['JSON_DB','Last Saved (IST)','Summary']); if (sh.getLastRow() > 1) sh.getRange(2,1,sh.getLastRow()-1,3).clearContent(); sh.getRange('A2').setValue(JSON.stringify(db)); sh.getRange('B2').setValue(new Date().toLocaleString('en-IN',{timeZone:'Asia/Kolkata'})); sh.getRange('C2').setValue( `Users:${(db.users||[]).length} | Att:${(db.attendance||[]).length} | Leaves:${(db.leaves||[]).length} | Ideas:${(db.ideas||[]).length} | Slips:${(db.salarySlips||[]).length}` ); sh.setColumnWidth(1, 60); // narrow — it's a blob sh.setColumnWidth(2, 200); sh.setColumnWidth(3, 360); } function _readRaw() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sh = ss.getSheetByName(RAW_SHEET); if (!sh) return null; const v = sh.getRange('A2').getValue(); return (v && v.toString().trim()) ? JSON.parse(v.toString()) : null; } function _tstStr(iso) { if (!iso) return ''; try { return new Date(iso).toLocaleString('en-IN',{timeZone:'Asia/Kolkata',day:'2-digit',month:'short',year:'numeric',hour:'2-digit',minute:'2-digit'}); } catch(e) { return String(iso).slice(0,19); } } function _dateStr(iso) { return iso ? String(iso).slice(0,10) : ''; } function _timeStr(iso) { if (!iso) return ''; try { return new Date(iso).toLocaleTimeString('en-IN',{timeZone:'Asia/Kolkata',hour:'2-digit',minute:'2-digit'}); } catch(e) { return ''; } } function _name(u) { return u ? ((u.first||'') + ' ' + (u.last||'')).trim() : ''; } function _userMap(users) { const m={}; (users||[]).forEach(u=>m[u.id]=u); return m; } function _writeUsers(users) { const H = ['Emp ID','Name','Role','Workflow','Team Leader','Join Date', 'Probation Months','Status','Confirmed On','Probation Note', 'Created','Pass Changed','Internal ID']; const sh = _sheet(SHEETS.users, H); _clearData(sh, H.length); if (!users||!users.length) return; const um = _userMap(users); const rows = users.map(u => [ u.empId||'', _name(u), u.role||'', u.workflow||'', um[u.tlId||''] ? _name(um[u.tlId])+'('+um[u.tlId].empId+')' : '', u.joinDate||_dateStr(u.createdAt), u.probationMonths||6, u.probationConfirmed ? '✅ Confirmed' : '⏳ On Probation', u.probationConfirmedAt ? _dateStr(u.probationConfirmedAt) : '', u.probationNote||'', _dateStr(u.createdAt), _dateStr(u.passChangedAt), u.id||'' ]); sh.getRange(2,1,rows.length,H.length).setValues(rows); rows.forEach((r,i) => { sh.getRange(i+2,8).setBackground(r[7].includes('Confirmed') ? '#dcfce7' : '#fef3c7'); }); } function _writeAttendance(attendance, users) { const H = ['Date','Emp ID','Name','Role','Punch In','Punch Out', 'Work Hours','Breaks','Break Time (min)','Status','Late?','Marked By']; const sh = _sheet(SHEETS.attendance, H); _clearData(sh, H.length); if (!attendance||!attendance.length) return; const um = _userMap(users); const sorted = [...attendance].sort((a,b)=>b.date.localeCompare(a.date)); const rows = sorted.map(a => { const u = um[a.userId]||{}; const mb = um[a.markedById||'']; const inT = a.punchIn ? new Date(a.punchIn) : null; const outT = a.punchOut ? new Date(a.punchOut) : null; const grossMs = (inT&&outT) ? outT-inT : 0; const workMs = Math.max(0, grossMs - (a.totalBreakMs||0)); const workH = workMs ? (Math.floor(workMs/3600000)+'h '+Math.floor((workMs%3600000)/60000)+'m') : ''; const breakMin= Math.round((a.totalBreakMs||0)/60000); return [ a.date||'', u.empId||'', _name(u), u.role||'', _timeStr(a.punchIn), _timeStr(a.punchOut), workH, (a.breaks||[]).length, breakMin, a.status||'present', a.lateMinutes>0 ? a.lateMinutes+'m late' : 'On time', mb ? _name(mb)+'('+mb.empId+')' : 'Self' ]; }); sh.getRange(2,1,rows.length,H.length).setValues(rows); } function _writeLeaves(leaves, users) { const H = ['Submitted','Emp ID','Name','Role','Type','Duration','From','To', 'Days','Status','Approver','Employee Reason','HR Reason', 'Salary Deduction','Decided On']; const sh = _sheet(SHEETS.leaves, H); _clearData(sh, H.length); if (!leaves||!leaves.length) return; const um = _userMap(users); const sorted = [...leaves].sort((a,b)=>(b.createdAt||'').localeCompare(a.createdAt||'')); const rows = sorted.map(l => { const u = um[l.userId]||{}; const approver = um[l.approverId]||{}; return [ _dateStr(l.createdAt), u.empId||'', _name(u), u.role||'', l.type||'', l.subType||'full', l.fromDate||'', l.toDate||'', l.leaveDays||0, l.status==='rejected'?'NOT APPROVED':(l.status||'').toUpperCase(), approver.empId ? _name(approver)+'('+approver.empId+')' : '', l.reason||'', l.hrRejectionReason||'', l.salaryDeduction?'YES':'NO', _dateStr(l.decidedAt) ]; }); sh.getRange(2,1,rows.length,H.length).setValues(rows); rows.forEach((r,i) => { const s = r[9]; sh.getRange(i+2,10).setBackground(s==='APPROVED'?'#dcfce7':s==='NOT APPROVED'?'#fee2e2':'#fef3c7'); if (r[13]==='YES') sh.getRange(i+2,14).setBackground('#fee2e2').setFontColor('#dc2626').setFontWeight('bold'); }); } function _writeIdeas(ideas, users) { const H = ['Submitted','Emp ID','Submitter','Title','Description', 'HR Status','Progress','HR Comment','Applied On']; const sh = _sheet(SHEETS.ideas, H); _clearData(sh, H.length); if (!ideas||!ideas.length) return; const um = _userMap(users); const rows = [...ideas].sort((a,b)=>(b.submittedAt||'').localeCompare(a.submittedAt||'')).map(i=>{ const u = um[i.userId]||{}; return [ _dateStr(i.submittedAt), u.empId||'', _name(u), i.title||'', i.description||'', (i.hrStatus||'pending').toUpperCase(), (i.progress||'initiated').replace('_',' ').toUpperCase(), i.hrComment||'', _dateStr(i.appliedAt) ]; }); sh.getRange(2,1,rows.length,H.length).setValues(rows); sh.setColumnWidth(5, 320); rows.forEach((r,i) => { const s = r[5]; sh.getRange(i+2,6).setBackground(s==='VALID'?'#dcfce7':s==='INVALID'?'#fee2e2':'#fef3c7'); }); } function _writeSalarySlips(slips, users) { const H = ['Created','Emp ID','Employee','Month','Net Pay (₹)','Basic','HRA', 'Other Allow','Deductions','Has PDF','Created By']; const sh = _sheet(SHEETS.salarySlips, H); _clearData(sh, H.length); if (!slips||!slips.length) return; const um = _userMap(users); const rows = slips.map(s => { const u = um[s.userId]||{}; const cb = um[s.createdBy||'']||{}; return [ _dateStr(s.createdAt), u.empId||'', _name(u), s.monthLabel||s.monthKey||'', s.netPay||0, s.basic||0, s.hra||0, s.otherAllowances||0, s.deductions||0, s.pdfData?'YES':'NO', cb.empId ? _name(cb)+'('+cb.empId+')' : '' ]; }); sh.getRange(2,1,rows.length,H.length).setValues(rows); } function _writeConfig(db) { const H = ['Section','Key','Value']; const sh = _sheet(SHEETS.config, H); _clearData(sh, H.length); const rows = []; (db.holidays||[]).forEach(h => rows.push(['Holiday', h.date, h.name])); (db.occasions||[]).forEach(o => rows.push(['Occasion', o.date, o.name+(o.desc?' — '+o.desc:'')])); Object.entries(db.shiftTimings||{}).forEach(([wf,s]) => rows.push(['Shift Timing', wf, 'Start: '+s.start+', Grace: '+s.grace+'min'])); rows.push(['System', 'Admin Code', db.adminCode||'']); rows.push(['System', 'Site Active', db.siteActive?'YES':'NO']); rows.push(['System', 'Chat Enabled', db.chatEnabled?'YES':'NO']); rows.push(['System', 'Location Lock', db.locationLock?.enabled?'Enabled':'Disabled']); rows.push(['System', 'Last Reset Year', db.lastResetYear||'']); rows.push(['System', 'Last Saved At', db._savedAt ? _tstStr(db._savedAt) : '']); rows.push(['System', 'Total Users', (db.users||[]).length]); rows.push(['System', 'Total Attendance', (db.attendance||[]).length]); rows.push(['System', 'Total Leaves', (db.leaves||[]).length]); rows.push(['System', 'Total Ideas', (db.ideas||[]).length]); if (rows.length) sh.getRange(2,1,rows.length,H.length).setValues(rows); } // ══════════════════ // GET — load DB // ══════════════════ function doGet(e) { try { const db = _readRaw(); return _cors(ContentService .createTextOutput(JSON.stringify({ ok: true, db: db })) .setMimeType(ContentService.MimeType.JSON)); } catch(err) { return _cors(ContentService .createTextOutput(JSON.stringify({ ok: false, error: err.message })) .setMimeType(ContentService.MimeType.JSON)); } } // ══════════════════ // POST — save / reset // ══════════════════ function doPost(e) { try { const body = JSON.parse(e.postData.contents); const action = body.action || 'save'; if (action === 'save') { const db = body.db; if (!db || !Array.isArray(db.users)) return _cors(ContentService .createTextOutput(JSON.stringify({ ok:false, error:'Invalid DB' })) .setMimeType(ContentService.MimeType.JSON)); _writeRaw(db); _writeUsers(db.users); _writeAttendance(db.attendance, db.users); _writeLeaves(db.leaves, db.users); _writeIdeas(db.ideas, db.users); _writeSalarySlips(db.salarySlips, db.users); _writeConfig(db); return _cors(ContentService .createTextOutput(JSON.stringify({ ok:true, savedAt: new Date().toISOString() })) .setMimeType(ContentService.MimeType.JSON)); } if (action === 'reset') { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sh = ss.getSheetByName(RAW_SHEET); if (sh && sh.getLastRow() > 1) sh.getRange(2,1,sh.getLastRow()-1,3).clearContent(); return _cors(ContentService .createTextOutput(JSON.stringify({ ok:true })) .setMimeType(ContentService.MimeType.JSON)); } return _cors(ContentService .createTextOutput(JSON.stringify({ ok:false, error:'Unknown action: '+action })) .setMimeType(ContentService.MimeType.JSON)); } catch(err) { return _cors(ContentService .createTextOutput(JSON.stringify({ ok:false, error:err.message })) .setMimeType(ContentService.MimeType.JSON)); } } // ═════════════════════════════ // TEST / SETUP UTILITIES // Run from Apps Script editor // ═════════════════════════════ function testSetup() { // Creates all sheets — run once after pasting to verify _sheet(RAW_SHEET, ['JSON_DB','Last Saved (IST)','Summary']); _sheet(SHEETS.leaves, ['Submitted','Emp ID','Name','Role','Type','Duration','From','To','Days','Status','Approver','Employee Reason','HR Reason','Salary Deduction','Decided On']); _sheet(SHEETS.ideas, ['Submitted','Emp ID','Submitter','Title','Description','HR Status','Progress','HR Comment','Applied On']);
